##Installation

    cd ~
    git clone git@github.com:pawelropa/dotvim.git ~/.dotvim
    ln -s ~/.dotvim/ ~/.vim
    ln -s ~/.dotvim/vimrc ~/.vimrc
    ln -s ~/.dotvim/gvimrc ~/.gvimrc
    ln -s ~/.dotvim/xvimrc ~/.xvimrc

##Adding submodules for plugins
    cd ~/.vim
    git submodule init
    git submodule update
